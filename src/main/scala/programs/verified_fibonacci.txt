
def fibonacci(n: Nat): Nat := {
    match n with {
        | 0 => 1
        | 1 => 1
        | i => fibonacci(i - 1) + fibonacci(i - 2)
    }
};

#![True] {
    let n := 5;
    let a := 0;
    let b := 1;
    while (n > 0) do {
        let b := a + b;
        let a := b - a;
        let n := n - 1
    };

    [a, fibonacci(5)]
} #![a = fibonacci(5)]
