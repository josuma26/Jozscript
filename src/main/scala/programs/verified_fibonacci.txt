
def fibonacci(n: Nat): Nat := {
    match n with {
        | 0 => 1
        | 1 => 1
        | i => fibonacci(i - 1) + fibonacci(i - 2)
    }
};

let N := 0;
#![True] {
    let n := N;
    let a := 1;
    let b := 1;
    while (n > 0) #![a = fibonacci(N - n) /\ b = fibonacci((N - n) + 1)] do {
        let b := a + b;
        let a := b - a;
        let n := n - 1
    };

    [a, fibonacci(N)]
} #![a = fibonacci(N)]
